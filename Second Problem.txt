#include <iostream>

using namespace std;

int main() {
    int nums[] = {1, 2, 3, 1, 1, 3};
    int count = 0;

    for (int i = 0; i < size(nums); i++) {
        for (int j = 0; j < size(nums); j++) {
            if (i < j) {
                if (nums[i] == nums[j]) {
                    // cout << "(" << i << "," << j << ")";
                    count++;
                }
            }
        }
    }

    return count;


    return 0;
}
